package nn;

public  class Least{
	
	public static void main(String[] args) {
		
		int[] a= {10,30,50,6,8,44,55};
		
		int max=0, secondmax=0;
		if(a[0]>a[1]) {
			max=a[0];
			secondmax=a[1];
		}
		else {
			max=a[1];
			secondmax=a[0];
		}
		for(int i=2;i<a.length;i++) {
			
			if(a[i]>max) {
				secondmax=max;
				max=a[i];
			}
			else if(a[i]>secondmax) {
				secondmax=a[i];
			}
		}
		System.out.println(secondmax);
		
		
	}
	
	
}
